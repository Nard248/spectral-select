Use the nbval approach:
- pytest --nbval-lax notebooks/examples/ runs notebooks ignoring output differences
- Mark cells to skip via # NBVAL_SKIP comment or cell metadata

Add pytest markers in pytest.ini:
- slow: marks tests as slow (deselect with '-m "not slow"')
- notebook: marks notebook tests

Pattern:
```python
import pytest
from pathlib import Path

NOTEBOOKS_DIR = Path(__file__).parent.parent / "notebooks" / "examples"

@pytest.mark.slow
@pytest.mark.notebook
class TestNotebookExecution:
    """Smoke tests for example notebooks."""

    def test_quickstart_cells_parseable(self):
        """Verify 01_quickstart.ipynb is valid JSON."""
        import json
        nb_path = NOTEBOOKS_DIR / "01_quickstart.ipynb"
        with open(nb_path) as f:
            nb = json.load(f)
        assert "cells" in nb
        assert len(nb["cells"]) > 0

    def test_validation_cells_parseable(self):
        """Verify 02_validation.ipynb is valid JSON."""
        import json
        nb_path = NOTEBOOKS_DIR / "02_validation.ipynb"
        with open(nb_path) as f:
            nb = json.load(f)
        assert "cells" in nb
```

def test_quickstart_imports():
